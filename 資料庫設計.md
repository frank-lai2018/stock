# 台股選股系統 — 資料庫設計（PostgreSQL + pgvector）

> 更新日：2026-07-04　｜　資料來源對應見 [資料盤點.md](資料盤點.md)、下載工具見 [README.md](README.md)
> 一庫到底：行情 / 基本面 / 籌碼估值 / 新聞向量(RAG) 全放 PostgreSQL。

---

## 設計原則

1. **Raw 層 + Serving 層**：財報照 FinMind 原樣存成「長表」(`financial_statement`) 保真；另建「寬表」(`fundamentals_quarterly`) 放選股常用指標，查詢好寫。
2. **Point-in-time 防前視**：每個「非當日就知道」的資料都存 `available_date`（可得日）。回測一律 `WHERE available_date <= 當日`。
   - 月營收：次月 10 日　｜　財報：Q1≈5/15、Q2≈8/14、Q3≈11/14、年報≈次年 3/31　｜　股利：用 FinMind 的 `AnnouncementDate`
3. **還原價同表存**：`price_daily` 同時存原始 OHLCV、還原 OHLCV、還原因子 `adj_factor`。技術面/回測一律讀 `adj_*`。
4. **單位正規化**：入庫時統一 → 成交量用「股」、金額用「元」。（TPEx 月檔是張/千元，載入時 ×1000 轉回股/元；三大法人淨額為股。）
5. **索引策略**：每張時序表主鍵 `(stock_id, trade_date)`（利於單檔時序）；另建 `trade_date` 索引（利於「某天全市場」橫斷面選股）。大型 append 表用 BRIN 索引省空間。
6. **代碼一致**：`stock_id VARCHAR(10)`，保留前導零（0050）；所有時序表對 `stock(stock_id)` 設外鍵。

## DDL（PostgreSQL 14+）

```sql
CREATE EXTENSION IF NOT EXISTS vector;   -- pgvector，供 RAG 用

-- ========== 維度 ==========
CREATE TABLE stock (
    stock_id    VARCHAR(10) PRIMARY KEY,   -- 股票代碼（保留前導零）
    name        TEXT NOT NULL,             -- 中文名稱
    market      VARCHAR(4),                -- 上市 / 上櫃
    list_date   DATE,                      -- 上市櫃日期
    industry    TEXT                       -- 產業別
);

-- ========== 行情 ==========
CREATE TABLE price_daily (
    stock_id    VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    trade_date  DATE        NOT NULL,
    open        NUMERIC(12,4),  high NUMERIC(12,4),  low NUMERIC(12,4),  close NUMERIC(12,4),
    volume      BIGINT,        -- 成交股數（股，已正規化）
    amount      BIGINT,        -- 成交金額（元，已正規化）
    trades      INTEGER,       -- 成交筆數
    adj_open    NUMERIC(12,4), adj_high NUMERIC(12,4), adj_low NUMERIC(12,4), adj_close NUMERIC(12,4),
    adj_factor  NUMERIC(16,10),-- 還原因子（cumfactor）
    PRIMARY KEY (stock_id, trade_date)
);
CREATE INDEX idx_price_date ON price_daily (trade_date);   -- 橫斷面（某天全市場）

CREATE TABLE market_index (
    index_id    VARCHAR(16) NOT NULL,      -- TAIEX 等
    trade_date  DATE        NOT NULL,
    close       NUMERIC(14,4),
    PRIMARY KEY (index_id, trade_date)
);

-- ========== 基本面 ==========
CREATE TABLE monthly_revenue (
    stock_id      VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    revenue_month DATE        NOT NULL,    -- 營收月份（該月1日）
    revenue       BIGINT,                  -- 當月營收（元）
    mom_pct       NUMERIC(8,2),            -- 月增率%
    yoy_pct       NUMERIC(8,2),            -- 年增率%
    available_date DATE,                   -- 可得日（次月10日）→ 回測用
    PRIMARY KEY (stock_id, revenue_month)
);

-- 長表：忠實保存 FinMind 損益/資產負債/現金流所有科目
CREATE TABLE financial_statement (
    stock_id     VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    period_date  DATE        NOT NULL,     -- 財報期別日（季底）
    statement    VARCHAR(10) NOT NULL,     -- income / balance / cashflow
    item         TEXT        NOT NULL,     -- FinMind 科目 key（如 Revenue, GrossProfit, EPS…）
    value        NUMERIC(20,4),
    available_date DATE,                   -- 可得日（財報公告慣例）
    PRIMARY KEY (stock_id, period_date, statement, item)
);
CREATE INDEX idx_fs_item ON financial_statement (stock_id, item, period_date);

-- 寬表：選股常用指標（由長表 + 股本 ETL 彙整）
CREATE TABLE fundamentals_quarterly (
    stock_id       VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    period_date    DATE        NOT NULL,
    available_date DATE,
    revenue          BIGINT,   gross_profit     BIGINT,
    operating_income BIGINT,   pretax_income    BIGINT,   net_income BIGINT,
    eps              NUMERIC(10,2),
    total_assets     BIGINT,   total_equity     BIGINT,   total_liabilities BIGINT,
    operating_cash_flow BIGINT,
    gross_margin   NUMERIC(8,2),  op_margin NUMERIC(8,2),  net_margin NUMERIC(8,2),
    roe            NUMERIC(8,2),  debt_ratio NUMERIC(8,2),          -- 由元件算
    PRIMARY KEY (stock_id, period_date)
);

CREATE TABLE dividend (
    stock_id       VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    announce_date  DATE        NOT NULL,   -- 公告日（point-in-time 基準）
    year_label     TEXT,                   -- FinMind year 欄（如「112年第3季」）
    cash_dividend  NUMERIC(10,6),          -- 現金股利合計（元/股）
    stock_dividend NUMERIC(10,6),          -- 股票股利合計（元/股）
    ex_cash_date   DATE,  ex_stock_date DATE,  cash_pay_date DATE,
    PRIMARY KEY (stock_id, announce_date)
);

CREATE TABLE capital_reduction (
    stock_id       VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    resume_date    DATE        NOT NULL,   -- 恢復買賣日
    pre_close      NUMERIC(12,4),          -- 停止買賣前收盤
    post_ref_price NUMERIC(12,4),          -- 恢復買賣參考價
    reason         TEXT,
    PRIMARY KEY (stock_id, resume_date)
);

-- ========== 籌碼 / 估值 ==========
CREATE TABLE inst_trades (            -- 三大法人買賣超（淨額，單位：股）
    stock_id     VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    trade_date   DATE        NOT NULL,
    foreign_net       BIGINT,         -- 外資
    foreign_dealer_net BIGINT,        -- 外資自營
    trust_net         BIGINT,         -- 投信
    dealer_self_net   BIGINT,         -- 自營商(自行買賣)
    dealer_hedge_net  BIGINT,         -- 自營商(避險)
    PRIMARY KEY (stock_id, trade_date)
);
CREATE INDEX idx_inst_date ON inst_trades (trade_date);

CREATE TABLE margin_trading (         -- 融資融券
    stock_id       VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    trade_date     DATE        NOT NULL,
    margin_balance BIGINT,            -- 融資今日餘額（張）
    short_balance  BIGINT,            -- 融券今日餘額（張）
    margin_buy     BIGINT, margin_sell BIGINT,
    short_sell     BIGINT, short_buy   BIGINT,
    PRIMARY KEY (stock_id, trade_date)
);

CREATE TABLE shareholding (           -- 外資持股 + 發行股數
    stock_id       VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    trade_date     DATE        NOT NULL,
    foreign_ratio  NUMERIC(8,2),      -- 外資持股比率%
    foreign_shares BIGINT,            -- 外資持股數
    shares_issued  BIGINT,            -- 發行股數（算市值/每股淨值用）
    PRIMARY KEY (stock_id, trade_date)
);

CREATE TABLE valuation_daily (        -- 每日估值
    stock_id       VARCHAR(10) NOT NULL REFERENCES stock(stock_id),
    trade_date     DATE        NOT NULL,
    per            NUMERIC(10,2),      -- 本益比
    pbr            NUMERIC(10,2),      -- 股價淨值比
    dividend_yield NUMERIC(8,2),       -- 殖利率%
    PRIMARY KEY (stock_id, trade_date)
);

-- ========== RAG（新聞 + 向量）==========
CREATE TABLE news (
    news_id      BIGSERIAL PRIMARY KEY,
    stock_id     VARCHAR(10) REFERENCES stock(stock_id),  -- 可為 NULL（總經新聞）
    title        TEXT,
    content      TEXT,
    source       TEXT,
    url          TEXT,
    published_at TIMESTAMPTZ            -- 發布時間（回測檢索防前視）
);
CREATE INDEX idx_news_stock_time ON news (stock_id, published_at);

CREATE TABLE news_embedding (
    news_id   BIGINT PRIMARY KEY REFERENCES news(news_id) ON DELETE CASCADE,
    model     TEXT,
    embedding vector(1536)             -- 依 embedding 模型維度調整
);
CREATE INDEX idx_news_vec ON news_embedding
    USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100);
```

## CSV → 表 載入對應（ETL 重點）

| 來源 | 目標表 | 轉換重點 |
|---|---|---|
| `<股號>_adj.csv` + 月檔 | `price_daily` | 民國→西元日期；TPEx 張/千元 **×1000 轉股/元**；併原始+還原欄 |
| `Index\TAIEX.csv` | `market_index` | `price`→`close` |
| `_revenue.csv` | `monthly_revenue` | `available_date` = 次月10日 |
| `_financials/_balance/_cashflow.csv` | `financial_statement` | 寬轉長（item, value）；`available_date` 依季別慣例 |
| （長表彙整）| `fundamentals_quarterly` | pivot 常用科目 + 算 margin/roe/debt |
| `_dividend.csv` | `dividend` | 現金/股票股利各自合計；`announce_date`=AnnouncementDate |
| `_capreduction.csv` | `capital_reduction` | 直接對應 |
| `_institutional.csv` | `inst_trades` | 各法人淨額欄對應 |
| `_margin.csv` | `margin_trading` | 取今日餘額等關鍵欄 |
| `_shareholding.csv` | `shareholding` | 取比率/持股/發行股數 |
| `_per.csv` | `valuation_daily` | PER/PBR/殖利率 |

## 常見查詢會長怎樣（設計驗證）

```sql
-- 某天全市場「均線多頭 + 外資連買 + 低本益比」候選（橫斷面選股）
SELECT p.stock_id, s.name, s.industry, v.per, i.foreign_net
FROM price_daily p
JOIN stock s USING (stock_id)
JOIN valuation_daily v USING (stock_id, trade_date)
JOIN inst_trades   i USING (stock_id, trade_date)
WHERE p.trade_date = DATE '2026-07-03'
  AND v.per < 15 AND i.foreign_net > 0;

-- 回測防前視：只用「當下已公告」的財報
SELECT * FROM fundamentals_quarterly
WHERE stock_id = '2330' AND available_date <= DATE '2023-06-01';
```
